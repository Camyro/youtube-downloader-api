    <script>
        // URL da sua API (altere para o endere√ßo do seu servidor)
        const API_URL = 'https://f1fb878c-7d3e-4ede-b87f-0654c776eb8c-00-3u0yl85htzmzl.riker.replit.dev';
        
        function showLoading() {
            document.getElementById('loading').classList.add('show');
            document.getElementById('error').classList.remove('show');
            document.getElementById('videoInfo').classList.remove('show');
        }
        
        function hideLoading() {
            document.getElementById('loading').classList.remove('show');
        }
        
        function showError(message) {
            const errorDiv = document.getElementById('error');
            errorDiv.textContent = message;
            errorDiv.classList.add('show');
            hideLoading();
        }
        
        function formatDuration(seconds) {
            const hours = Math.floor(seconds / 3600);
            const minutes = Math.floor((seconds % 3600) / 60);
            const secs = seconds % 60;
            
            if (hours > 0) {
                return `${hours}h ${minutes}m ${secs}s`;
            }
            return `${minutes}m ${secs}s`;
        }
        
        async function getVideoInfo() {
            const url = document.getElementById('videoUrl').value;
            
            if (!url) {
                showError('Por favor, cole uma URL do YouTube');
                return;
            }
            
            showLoading();
            
            try {
                const response = await fetch(`${API_URL}/info?url=${encodeURIComponent(url)}`);
                const data = await response.json();
                
                if (!response.ok) {
                    throw new Error(data.error || 'Erro ao buscar informa√ß√µes');
                }
                
                // Mostrar informa√ß√µes do v√≠deo
                document.getElementById('thumbnail').src = data.thumbnail;
                document.getElementById('title').textContent = data.title;
                document.getElementById('uploader').textContent = data.uploader;
                document.getElementById('duration').textContent = formatDuration(data.duration);
                
                document.getElementById('videoInfo').classList.add('show');
                hideLoading();
                
            } catch (error) {
                showError('Erro: ' + error.message);
            }
        }
        
        async function downloadVideo() {
            const url = document.getElementById('videoUrl').value;
            
            if (!url) {
                showError('Por favor, cole uma URL do YouTube');
                return;
            }
            
            showLoading();
            
            try {
                // Criar link de download
                const downloadUrl = `${API_URL}/download?url=${encodeURIComponent(url)}`;
                
                // Abrir em nova aba para iniciar o download
                window.open(downloadUrl, '_blank');
                
                hideLoading();
                
            } catch (error) {
                showError('Erro ao baixar: ' + error.message);
            }
        }

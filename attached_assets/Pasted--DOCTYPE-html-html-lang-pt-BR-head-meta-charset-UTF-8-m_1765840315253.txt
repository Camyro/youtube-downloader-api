    <script>
        // ALTERE PARA A URL DA SUA API
        const API_URL = 'https://youtube-downloader-api-1-lgmy.onrender.com';
        
        let currentUrl = '';
        
        function showLoading() {
            document.getElementById('loading').classList.add('show');
            document.getElementById('error').classList.remove('show');
            document.getElementById('videoPreview').classList.remove('show');
        }
        
        function hideLoading() {
            document.getElementById('loading').classList.remove('show');
        }
        
        function showError(message) {
            const errorDiv = document.getElementById('error');
            errorDiv.textContent = '‚ùå ' + message;
            errorDiv.classList.add('show');
            hideLoading();
        }
        
        function formatDuration(seconds) {
            const hours = Math.floor(seconds / 3600);
            const minutes = Math.floor((seconds % 3600) / 60);
            const secs = seconds % 60;
            
            if (hours > 0) {
                return `${hours}h ${minutes}m ${secs}s`;
            }
            return `${minutes}m ${secs}s`;
        }
        
        function formatNumber(num) {
            if (!num) return 'N/A';
            return num.toLocaleString('pt-BR');
        }
        
        function formatDate(dateStr) {
            if (!dateStr) return 'N/A';
            // PyTube retorna ISO format (2024-12-15T00:00:00)
            try {
                const date = new Date(dateStr);
                return date.toLocaleDateString('pt-BR');
            } catch {
                return 'N/A';
            }
        }
        
        function formatFileSize(bytes) {
            if (!bytes) return 'N/A';
            const mb = (bytes / (1024 * 1024)).toFixed(2);
            return `${mb} MB`;
        }
        
        async function loadVideo() {
            const url = document.getElementById('videoUrl').value.trim();
            
            if (!url) {
                showError('Por favor, cole uma URL v√°lida');
                return;
            }
            
            currentUrl = url;
            showLoading();
            
            try {
                const response = await fetch(`${API_URL}/info?url=${encodeURIComponent(url)}`);
                const data = await response.json();
                
                if (!response.ok) {
                    throw new Error(data.error || 'Erro ao buscar informa√ß√µes');
                }
                
                // Preencher informa√ß√µes
                document.getElementById('thumbnail').src = data.thumbnail;
                document.getElementById('title').textContent = data.title;
                document.getElementById('uploader').textContent = data.uploader || data.author;
                document.getElementById('duration').textContent = data.duration_string || formatDuration(data.duration);
                document.getElementById('views').textContent = formatNumber(data.view_count);
                document.getElementById('date').textContent = formatDate(data.publish_date || data.upload_date);
                
                // Mostrar preview
                document.getElementById('videoPreview').classList.add('show');
                hideLoading();
                
            } catch (error) {
                showError(error.message);
            }
        }
        
        function downloadBest() {
            if (!currentUrl) return;
            const downloadUrl = `${API_URL}/download?url=${encodeURIComponent(currentUrl)}&quality=highest`;
            window.open(downloadUrl, '_blank');
        }
        
        function downloadAudio() {
            if (!currentUrl) return;
            const downloadUrl = `${API_URL}/download-audio?url=${encodeURIComponent(currentUrl)}`;
            window.open(downloadUrl, '_blank');
        }
        
        function download1080p() {
            if (!currentUrl) return;
            const downloadUrl = `${API_URL}/download?url=${encodeURIComponent(currentUrl)}&quality=1080p`;
            window.open(downloadUrl, '_blank');
        }
        
        function download720p() {
            if (!currentUrl) return;
            const downloadUrl = `${API_URL}/download?url=${encodeURIComponent(currentUrl)}&quality=720p`;
            window.open(downloadUrl, '_blank');
        }
        
        function download480p() {
            if (!currentUrl) return;
            const downloadUrl = `${API_URL}/download?url=${encodeURIComponent(currentUrl)}&quality=480p`;
            window.open(downloadUrl, '_blank');
        }
        
        async function toggleFormats() {
            const formatsList = document.getElementById('formatsList');
            
            if (formatsList.classList.contains('show')) {
                formatsList.classList.remove('show');
                return;
            }
            
            if (formatsList.innerHTML) {
                formatsList.classList.add('show');
                return;
            }
            
            // Carregar formatos
            try {
                const response = await fetch(`${API_URL}/formats?url=${encodeURIComponent(currentUrl)}`);
                const data = await response.json();
                
                if (!response.ok) {
                    throw new Error(data.error || 'Erro ao buscar formatos');
                }
                
                let html = '';
                data.formats.forEach(fmt => {
                    const resolution = fmt.resolution || 'N/A';
                    const filesize = fmt.filesize_mb ? `${fmt.filesize_mb} MB` : 'N/A';
                    const hasVideo = fmt.includes_video ? 'üé¨ V√≠deo' : '';
                    const hasAudio = fmt.includes_audio ? 'üéµ √Åudio' : '';
                    const media = [hasVideo, hasAudio].filter(Boolean).join(' + ') || 'N/A';
                    
                    html += `
                        <div class="format-item">
                            <div class="format-info">
                                <strong>${resolution} - ${fmt.mime_type || 'N/A'}</strong>
                                <span>
                                    ${filesize} | ${media} | ${fmt.fps ? fmt.fps + 'fps' : ''}
                                </span>
                            </div>
                            <button class="format-download" onclick="downloadFormat('${fmt.itag}')">
                                Baixar
                            </button>
                        </div>
                    `;
                });
                
                formatsList.innerHTML = html;
                formatsList.classList.add('show');
                
            } catch (error) {
                showError(error.message);
            }
        }
        
        function downloadFormat(formatId) {
            if (!currentUrl) return;
            const downloadUrl = `${API_URL}/download?url=${encodeURIComponent(currentUrl)}&format=${formatId}`;
            window.open(downloadUrl, '_blank');
        }
